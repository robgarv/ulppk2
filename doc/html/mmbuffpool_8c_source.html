105 int main(int argc, char* argv[]) {
106  int status;
107  int require_pooldir = 0;
108  char* pooldir = NULL;
109  char* overridedir = NULL;
110 
111  // Register the MMPOOL_ENV_DATA_DIR variable
112  // in the application environment
113  pooldir = appenv_register_env_var(MMPOOL_ENV_DATA_DIR, "/var/ulppk/data");
114  if (! pathinfo_is_dir(pooldir)) {
115  // The path specified by environment variable MMPOOL_ENV_DATA_DIR
116  // does exist. Either a) the variable is not defined in the calling
117  // shell environment and the default value /var/ulppk/data is incorrect
118  // or b) the environment variable was set incorrectly. CHECK PERMISSIONS
119  // because pathinfo_is_dir can be deceived by a directory it does not
120  // have permission to access. In any case, we now NEED the -d option
121  // in the command line.
122  require_pooldir = 1;
123  }
124  register_args(argc, argv);
125 
126  status = cmdarg_parse(argc, argv);
127  if (status) {
128  cmdarg_show_help(NULL);
129  exit(1);
130  }
131 
132  // Parsed arguments successfully ... process. First,
133  // check is user is just asking for help. If so, program
134  // will print help and exit.
135  status = process_switch_help();
136 
137  // Then get the pool directory override (if any). Note fetch_pool_dir
138  // can return the original environment pool dir is -d
139  // is not provided or is incorrect.
140  overridedir = fetch_pool_dir(pooldir, require_pooldir);
141  appenv_set_env_var(MMPOOL_ENV_DATA_DIR, overridedir);
142 
143  // Handler routines
144  // return > 0 if successful, 0 if they took no action, and < 0
145  // if they encountered an error.
146  if (0 == status) {
147  status = process_switch_create();
148  }
149  if (0 == status) {
150  status = process_switch_report();
151  }
152  if (0 == status) {
153  status = process_switch_display();
154  }
155  if (0 == status) {
156  status = process_switch_zap();
157  }
158  if (status > 0) {
159  status = 0;
160  }
161  return status;
162 }

208 static char* fetch_pool_dir(char* pool_dir, int dir_required) {
209  char* pool_dir_override;
210  char* outpooldir;
211 
212  pool_dir_override = cmdarg_fetch_string(NULL, "d");
213 
214  if (dir_required) {
215  if ((pool_dir_override == NULL ) || (strlen(pool_dir_override) == 0)) {
216  // Command line did not give us a directory path for the pool data files
217  // and neither did the environment ... can't go on
218  app_error("Environment does not provide valid MMPOOL_ENV_DATA_DIR setting: -d option needed");
219  } else {
220  if (pathinfo_is_dir(pool_dir_override)) {
221  outpooldir = pool_dir_override;
222  } else {
223  app_error("-d option did not provide accessible directory path");
224  }
225  }
226  } else {
227  // The environment gave us a good path but that can be overriden
228  // if -d option provides a valid data path
229  if ((pool_dir_override == NULL ) || (strlen(pool_dir_override) == 0)) {
230  // Command line did not give us a directory path for the pool data files
231  // but environment did ... use environment
232  outpooldir = pool_dir;
233  } else {
234  if (pathinfo_is_dir(pool_dir_override)) {
235  outpooldir = pool_dir_override;
236  } else {
237  outpooldir = pool_dir;
238  }
239  }
240  }
241  return outpooldir;
242 }

365 static int display_pool(BPOOL_HANDLE* bphp) {
366  DQHEADER* dqp;
367  DQHEADER tempdq;
368  int buffcount = 0;
369  BPOOL_INDEX bpx = 0;
370 
371  // TODO: lock
372  dqp = &bphp->bpmf_recp->dq_inpool;
373 
374  // Construct a temporary deque header
375  dq_init(dqp->dqslots, dqp->dqitem_size, &tempdq);
376 
377  for (bpx = 0; (!dq_rtd(dqp, &bpx)); ) {
378  buffcount++;
379  dq_abd(&tempdq, &bpx); // push onto the temporary deque
380  rptline("Rec: %d BPX: %ld", buffcount, bpx);
381  }
382  if (buffcount != bphp->bpmf_recp->stats.remaining) {
383  rptline("WARNING: Header balance problem! Buffcount = %d stats.remaining = %d",
384  buffcount, bphp->bpmf_recp->stats.remaining);
385  }
386  // Restore the memory mappe deque to its original state
387  for (bpx = 0; (!dq_rtd(&tempdq, &bpx)); ) {
388  dq_abd(dqp, &bpx);
389  }
390  // TODO: Unlock
391  return buffcount;
392 }
