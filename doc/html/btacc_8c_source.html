231  void bt_del(PTREE_NODE *tree,PTREE_NODE node){
232 
233  PTREE_NODE parent; /* pointer to a parent node */
234  PTREE_NODE repnode; /* ptr to node chosen to replace node to delete */
235  PTREE_NODE subtree; /* ptr to a subtree hanging off node to delete */
236 
237 
238  /* BEGIN */
239 
240  if (*tree == NULL) { /* asked to remove from empty tree */
241  return;
242  }
243 
244  /*
245  * Deleting an arbitrary node without massive re-arrangement of the tree
246  * depends on the following observation:
247  * IF there is a left subtree depending on the node, obtaining the
248  * rightmost leaf of that subtree provides a node whose tag
249  * value is between those of the current left and right children
250  * of the node we are deleting. Call this node the "replacement node".
251  * (It thus has a value that suits it to occupy the position of the
252  * node we are deleting.)
253  * IF there is a left subtree to the replacement node, its left
254  * child takes its place as the right child of the replacement
255  * node's parent.
256  * ELSE IF there is a right subtree, obtaining the leftmost leaf
257  * of that subtree produces a similar node.
258  * IF there is a right subtree to the replacement node, its
259  * right child takes its place as the left child of the replacement
260  * node's parent.
261  * ELSE no subtrees dependent on the node to be deleted exist,
262  * our job is childlike in simplicity in that all we have to
263  * do is NULL out the appropriate linkage in the parent.
264  */
265 
266  if (node->tr_left != NULL) { /* we have a left subtree */
267  subtree = node->tr_left; /* get a pointer to it */
268  repnode= bt_rightmost(subtree); /* get the rightmost as replacement */
269  /*
270  * If repnode is NULL, then the subtree node is the rightmost
271  * node of the left subtree, and thus becomes the replacement
272  * node.
273  */
274  if (repnode == NULL) {
275  repnode = subtree;
276  }
277  /*
278  * We are at the rightmost node of the subtree left of the node to be
279  * removed. The left child of this node (repnode) becomes the right
280  * child of repnode's parent. Note the left child may be NULL if the
281  * is no left subtree hanging off repnode. (In other words, if it has one,
282  * repnode's left subtree becomes its parent's right subtree.) HOWEVER
283  * If repnode's parent is node (the node we're deleting), then repnode
284  * must retain its left subtree.
285  */
286  parent = repnode->tr_parent; /* get pointer to its parent */
287  if (parent != node) { /* adjust parent's right linkage */
288  parent->tr_right = repnode->tr_left; /* parents right becomes repnodes left */
289  if (parent->tr_right != NULL) {
290  parent->tr_right->tr_parent = parent; /* adjust child's parent ptr */
291  }
292  }
293  /*
294  * Now replace node with repnode ... step 1 adjust node's parent's
295  * linkages. (node's parent becomes repnode's parent)
296  */
297  if (node->tr_parent == NULL) { /* replacing the root node */
298  *tree = repnode; /* adjust root ptr accordingly */
299  } else {
300  if (node->tr_parent->tr_left == node) {
301  node->tr_parent->tr_left = repnode;
302  } else {
303  node->tr_parent->tr_right = repnode;
304  }
305  }
306  /*
307  * Step 2: adjust repnode's linkages. Repnode inherits the left and right
308  * subtrees of node (the node we are deleting). However, repnode may
309  * be the left child of node, in which case we must leave its left pointer
310  * alone.
311  */
312  repnode->tr_parent = node->tr_parent; /* write parent link */
313  if (node->tr_left != repnode) { /* repnode is left child of node */
314  repnode->tr_left = node->tr_left; /* repnode gets node's left subtree */
315  }
316  repnode->tr_right = node->tr_right; /* write the right child link */
317  /*
318  * Step 3: Adjust parent linkages of repnode's new children
319  */
320  if (repnode->tr_right != NULL) {
321  repnode->tr_right->tr_parent = repnode;
322  }
323  if (repnode->tr_left != NULL) {
324  repnode->tr_left->tr_parent = repnode;
325  }
326  } else if (node->tr_right != NULL) { /* at least we have a right subtree */
327  subtree = node->tr_right; /* get a pointer to it */
328  repnode= bt_leftmost(subtree); /* get the leftmost member */
329  /*
330  * If repnode is NULL, then the subtree node is the leftmost
331  * node of the right subtree, and thus becomes the replacement
332  * node.
333  */
334  if (repnode == NULL) {
335  repnode = subtree;
336  }
337  /*
338  * We are at the leftmost node of the subtree right of the node to be
339  * removed. The right child of this node (repnode) becomes the left
340  * child of repnode's parent. Note the right child may be NULL if there
341  * is no right subtree hanging off repnode. (In other words, if it has one,
342  * repnode's right subtree becomes its parent's left subtree.) HOWEVER
343  * If repnode's parent is node (the node we're deleting), then repnode
344  * must retain its right subtree.
345  */
346  parent = repnode->tr_parent; /* get pointer to its parent */
347  if (parent != node) { /* adjust parent's left linkage */
348  parent->tr_left = repnode->tr_right; /* change repnode's parent */
349  if (parent->tr_left != NULL) {
350  parent->tr_left->tr_parent = parent; /* change child's parent ptr */
351  }
352  }
353  /*
354  * Now replace node with repnode ... step 1 adjust node's parent's
355  * linkages. (node's parent becomes repnode's parent)
356  */
357  if (node->tr_parent == NULL) { /* replaced the root node */
358  *tree = repnode; /* adjust root ptr accordingly */
359  } else {
360  if (node->tr_parent->tr_left == node) {
361  node->tr_parent->tr_left = repnode;
362  } else {
363  node->tr_parent->tr_right = repnode;
364  }
365  }
366  /*
367  * Step 2: adjust repnode's linkages. Repnode inherits the left and right
368  * subtrees of node (the node we are deleting). However, repnode may
369  * be the right child of node, in which case we must leave its right pointer
370  * as it is.
371  */
372  repnode->tr_parent = node->tr_parent; /* write parent link */
373  if (node->tr_right != repnode) { /* repnode is right child of node */
374  repnode->tr_right = node->tr_right; /* repnode gets node's right subtree */
375  }
376  repnode->tr_left = node->tr_left;
377  /*
378  * Step 3: Adjust parent linkages of repnode's new children
379  */
380  if (repnode->tr_right != NULL) {
381  repnode->tr_right->tr_parent = repnode;
382  }
383  if (repnode->tr_left != NULL) {
384  repnode->tr_left->tr_parent = repnode;
385  }
386  } else {
387  /*
388  * The node to removed is a leaf. Life is suddenly simple. Were it so always ...
389  */
390  parent = node->tr_parent; /* get pointer to parent */
391  if (parent == NULL) { /* Wow! deleting only node in tree */
392  *tree = NULL; /* Null out root pointer */
393  } else { /* Deleting leaf of a subtree */
394  if (parent->tr_left == node) { /* we're left child of parent */
395  parent->tr_left = NULL; /* NULL left child link of parent */
396  } else { /* must be the right child */
397  parent->tr_right = NULL; /* so NULL right child link of parent */
398  }
399  }
400  }
