119 char* appenv_register_env_var(const char* varname, const char* defaultval) {
120  ENV_NODE* varnodep = NULL;
121  char* env_valp = NULL;
122  int add_node = 0;
123 
124  // First, determine if the variable has already been stored in a
125  // ENV_NODE in the tree.
126  if (NULL != root.rootp) {
127  // Tree is not empty ... do search
128  varnodep = search_tree((char*)varname);
129 
130  }
131  if (NULL == varnodep) {
132  add_node = 1;
133  varnodep = (ENV_NODE*)calloc(1, sizeof(ENV_NODE));
134  }
135 
136  // Now check the environment to see if the variable has been set
137  // by the invoking shell
138  env_valp = getenv(varname);
139 
140  // Write the environment node. If env_val is NULL ... enforce the default
141  // if provided. If no default was provided, enforce the compile time
142  // default value
143  strncpy(varnodep->varname, varname, sizeof(varnodep->varname) - 1);
144  if (NULL == env_valp) {
145  if (defaultval != NULL) {
146  strncpy(varnodep->value, defaultval, sizeof(varnodep->value) - 1);
147  } else {
148  strcpy(varnodep->value, "");
149  }
150  } else {
151  strncpy(varnodep->value, env_valp, sizeof(varnodep->value) - 1) ;
152  }
153 
154  if (add_node) {
155  // Add the variable to the tree
156  bt_add(((PTREE_NODE*)&root.rootp), varnodep, varname_cmpfunc);
157  }
158  return varnodep->value;
159 }
