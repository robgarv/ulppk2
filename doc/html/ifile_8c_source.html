338  INIFILE_NODE* sectionp;

341  sectionp = (INIFILE_NODE*)tnodep;
342  if (argbuff == NULL) {
343  outfilep = stdout;
344  } else {
345  outfilep = (FILE*)argbuff;
346  }
347  if (sectionp->node_type != IFNT_SECTION) {
348  fprintf(stderr, "Fatal Error: Exepcted a SECTION Node!\n");
349  fprintf(stderr, "INIFILE binary tree structures are corrupt\n");
350  return 0; // indicate traversal abort
351  }
352 
353  // Perform in-order traversal of the tree of elements
354  fprintf(outfilep, "SECTION|%s|\n", sectionp->name);
355

357  bt_inorder(sectionp->section.elements, argbuff, print_element);
358  }
359  return 1; // indicate traversal continue
360 }

362 static int print_element(TREE_NODE* tnodep, void* argbuff) {
363  INIFILE_NODE* elementp;
364  FILE* outfilep;
365 
366  elementp = (INIFILE_NODE*)tnodep;
367  if (argbuff == NULL) {
368  outfilep = stdout;
369  } else {
370  outfilep = (FILE*)argbuff;
371  }
372 
373  if (elementp->node_type != IFNT_ELEMENT) {
374  fprintf(stderr, "Fatal Error: Exepcted a ELEMENT Node!\n");
375  fprintf(stderr, "INIFILE binary tree structures are corrupt\n");
376  return 0; // indicate traversal abort
377  }
378 
379  fprintf(outfilep, "ELEMENT|%s|VALUE|", elementp->name);
380  switch (elementp->element.data_type) {
381  case IFDT_STRING:
382  fprintf(outfilep, "%s|\n", elementp->element.szvalue);
383  break;
384  case IFDT_NUMBER:
385  fprintf(outfilep,"%ld|%lx|\n", elementp->element.lvalue, elementp->element.lvalue);
386  break;
387  default:
388  fprintf(stderr, "Fatal Error: Undefined ELEMENT Node Data Type!\n");
389  fprintf(stderr, "INIFILE binary tree structures are corrupt\n");
390  return 0; // indicate traversal abort
391  }
392  return 1;
393 }
