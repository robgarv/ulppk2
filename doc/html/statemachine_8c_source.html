367 static SM_ACTION* find_action(SM_MACHINE *machinep, const char * actlist_name,
368  const char *action_name) {
369  SM_ACTION_LIST_HANDLE alh = SM_NULL_HANDLE;
370  SM_STATE_TABLE_DEF* stdefp;
371  LL_HEAD* llhp;
372  SM_ACTION* actp = NULL;
373  LL_NODE* nodep;
374 
375  stdefp = get_stdefp(machinep);
376 
377  // Find the action list named by the caller.
378  alh = find_action_list(machinep, actlist_name);
379  if (SM_NULL_HANDLE == alh) {
380  return NULL; // action list not found!
381  }
382  llhp = &stdefp->actlists[alh].list;
383  nodep = ll_search(llhp, (char*)action_name, match_actname);
384  if (nodep != NULL) {
385  actp = (SM_ACTION*)nodep->data;
386  }
387  return actp;
388 }

412  stdefp->next_eventh++;

416  }

464 SM_ACTION_HANDLE sm_register_action(SM_MACHINE* machinep, const char* actlist_name,
465  const char* name, SM_ACTION_HANDLER* handlerp) {
466 
467  SM_ACTION_LIST_HANDLE alh;
468  SM_STATE_TABLE_DEF* stdefp;
469  LL_HEAD *llhp;
470  SM_ACTION act;
471  LL_NODE *nodep;
472  SM_ACTION_HANDLE acth = SM_NULL_HANDLE;
473 
474  clear_err(machinep);
475 
476  stdefp = get_stdefp(machinep);
477 
478  alh = find_action_list(machinep, actlist_name);
479  if (SM_NULL_HANDLE == alh) {
480  set_diag(machinep, SM_PARENT_OBJECT_NOT_FOUND, actlist_name);
481  return SM_NULL_HANDLE;
482  }
483  llhp = &stdefp->actlists[alh].list;
484 
485  // Determine if an action of this name is already registered
486  // to this list. Actions can be registered to more than one list,
487  // but not to the same list more than once.
488 
489  if (find_action(machinep, actlist_name, name) == NULL) {
490  strncpy(act.action_name, name, sizeof(act.action_name) - 1);
491  act.actionh = stdefp->next_acth++;
492  act.handlerp = handlerp;
493 
494  // ll_new_node makes a copy of the passed buffer (act) and installs
495  // it in a fresh ll_node ready for list insertion.
496  nodep = ll_new_node(&act, sizeof(SM_ACTION));
497  ll_addback(llhp, nodep);
498  acth = act.actionh;
499  } else {
500  set_diag(machinep, SM_REGISTER_DUPLICATE, name);
501  }
502  return acth;
503 }

549 SM_TRANSITION_HANDLE sm_register_transition(
550  SM_MACHINE *machinep, // state machine
551  const char *s1name, // name of the initial state
552  const char *s2name, // name of the final state
553  const char* evname, // name of the triggering event
554  const char* actlist_name // name of the action list to execute
555 ) {
556  SM_STATE_HANDLE st1h;
557  SM_STATE_HANDLE st2h;
558  SM_EVENT_HANDLE evh;
559  SM_ACTION_LIST_HANDLE alh;
560  SM_TRANSITION_HANDLE stxh;
561  SM_TRANSITION* txp;
562  SM_STATE_TABLE_DEF* stdefp;
563 
564  stdefp = get_stdefp(machinep);
565 
566  // Find initial and final states
567  st1h = find_state(machinep, s1name);
568  st2h = find_state(machinep, s2name);
569 
570  // Find triggering event
571  evh = find_event(machinep, evname);
572 
573  // Find the action list
574  alh = find_action_list(machinep, actlist_name);
575 
576  if (st1h == SM_NULL_HANDLE) {
577  set_diag(machinep, SM_OBJECT_NOT_FOUND, s1name);
578  return SM_NULL_HANDLE;
579  }
580  if (st2h == SM_NULL_HANDLE) {
581  set_diag(machinep, SM_OBJECT_NOT_FOUND, s2name);
582  return SM_NULL_HANDLE;
583  }
584  if (evh == SM_NULL_HANDLE) {
585  set_diag(machinep, SM_OBJECT_NOT_FOUND, evname);
586  return SM_NULL_HANDLE;
587  }
588  if (alh == SM_NULL_HANDLE) {
589  set_diag(machinep, SM_OBJECT_NOT_FOUND, actlist_name);
590  return SM_NULL_HANDLE;
591  }
592 
593  // If we get here, all the named objects where found. Register the
594  // transition if we have room.
595  if (stdefp->states[st1h].next_txh >= MAX_TRANSITIONS_PER_STATE) {
596  set_diag(machinep, SM_TOO_MANY_OBJECTS,
597  make_transition_name(NULL, s1name, evname, s2name));
598  return SM_NULL_HANDLE;
599  }
600  stxh = stdefp->states[st1h].next_txh;
601  stdefp->states[st1h].next_txh++;
602  txp = &stdefp->states[st1h].transitions[stxh];
603  txp->alisth = alh;
604  txp->eventh = evh;
605  txp->s1 = st1h;
606  txp->s2 = st2h;
607  make_transition_name(txp->transition_name, s1name, evname, s2name);
608  return stxh;
609 }

742 SM_ERROR sm_transition(SM_MACHINE* machinep, const char* event_name, void* datap) {
743  SM_EVENT_HANDLE evh = SM_NULL_HANDLE;
744  SM_EVENT_HANDLE queue_evh = SM_NULL_HANDLE;
745  SM_TRANSITION* txp;
746  SM_STATE_TABLE_DEF* stdefp;
747  SM_TRANSITION_HANDLE txh;

769  if (!sm_get_definition_complete(machinep)) {

772  return SM_MACHINE_DEF_NOT_COMPLETE;
773  }
774 
775  state1p = &stdefp->states[machinep->curr_state]; // point to current state
776 
777  // Now select transition
778  for (txh = 0, txp = NULL; txh < state1p->next_txh; txh++) {
779  if (state1p->transitions[txh].eventh == evh) {
780  // Found the matching state transition ...
781  // Execute the action list
782  txp = &state1p->transitions[txh];
783  llhp = &stdefp->actlists[txp->alisth].list; // get ptr to linked list
784  nodep = llhp->ll_fptr; // point to first node on list
785  if (!is_named_event(machinep, evh, EV_CLOCK)) {
786  // Don't log clock pulses
787  SMLOG(machinep, "S1 [%s] E [%s] S2 [%s] AL [%s]",
788  state1p->state_name, event_name,
789  stdefp->states[txp->s2].state_name,
790  &stdefp->actlists[txp->alisth].al_name);
791  }
792  while (nodep != NULL) {
793  SM_ACTION* actp;
794  SM_EVENT_HANDLE action_evh = SM_NULL_HANDLE;
795 
796  actp = (SM_ACTION*)nodep->data;
797  action_evh = actp->handlerp(machinep, datap);
798  if (!is_named_event(machinep, evh, EV_CLOCK)) {
799  // Don't log clock pulse actions ..
800  SMLOG(machinep, "\tACTION: %s ==> Event %s",
801  actp->action_name, stdefp->events[action_evh].event_name);
802  }
803  dq_abd(&machinep->event_queue, &action_evh);
804  nodep = ll_iterate(nodep);
805  }
806  // Now set new machine state
807  machinep->prev_state = machinep->curr_state;
808  machinep->curr_state = txp->s2;
809  machinep->last_event = txp->eventh;
810  break; // break out of the for
811  }
812  }
813 
814  // Process any events stacked up while processing the action list
815  // This necessitates a recursive call to sm_transition_machine!
816  while ((status == SM_NO_ERROR) && (!dq_rtd(&machinep->event_queue, &queue_evh))) {
817  status = sm_transition(machinep,
818  stdefp->events[queue_evh].event_name, datap);
819  }
820  if (SM_NO_ERROR != status) {
821  set_diag(machinep, status, machinep->name);
822  push_txdiag(machinep, status, queue_evh);
823  }
824  return status;
825 }
