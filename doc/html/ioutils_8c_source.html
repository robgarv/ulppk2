65 /*
66  * Write n bytes to a descriptor
67  * @param fd File descriptor of open file
68  * @param vptr void pointer to area with data to write
69  * @param n number of bytes to write out of vptr
70  * @return Number of bytes read.
71  */
72 size_t ioutils_writen(int fd, const void *vptr, size_t n) {
73  size_t nleft, nwritten;
74  const char *ptr;
75 
76  ptr = vptr; /* can't do pointer arithmetic on void* */
77  nleft = n;
78  while (nleft > 0) {
79  if ( (nwritten = write(fd, ptr, nleft)) <= 0)
80  return(nwritten); /* error */
81 
82  nleft -= nwritten;
83  ptr += nwritten;
84  }
85  return(n);
86 }

88 /*
89  * Read n bytes from a descriptor
90  *
91  * @param fd File descriptor of open file
92  * @param vptr void pointer to area to receive data
93  * @param n Max number of bytes to read into vptr
94  * @return Number of bytes read.
95  */
96 size_t ioutils_readn(int fd, void *vptr, size_t n)
97 {
98  size_t nleft, nread;
99  char *ptr;
100 
101  ptr = vptr;
102  nleft = n;
103  while (nleft > 0) {
104  if ( (nread = read(fd, ptr, nleft)) < 0)
105  return(nread); /* error, return < 0 */
106  else if (nread == 0)
107  break; /* EOF */
108 
109  nleft -= nread;
110  ptr += nread;
111  }
112  return(n - nleft); /* return >= 0 */
113 }

128 size_t ioutils_readln(int fd, char* buff, size_t buffsize) {
129  int i;
130  int eol;
131  size_t totalread;
132  size_t n2read;
133  size_t nread;
134  char* op;
135 
136  totalread = 0;
137  eol = 0;
138  n2read = buffsize - 1;
139 
140  for (i = 0, op = buff; ((i < n2read) && (!eol)); i++) {
141  nread = read(fd, op, 1);
142  if (nread < 0) {
143  return totalread;
144  } else if (nread > 0) {
145  totalread++;
146  if (*op == '\r') {
147  // \r\n DOS line termination 1st char
148  // Just skip it
149  continue;
150  } else if (*op == '\n') {
151  // End of \r\n or \n line termination
152  // Null terminate and bail
153  *op = '\0';
154  op++;
155  return totalread;
156  } else {
157  op++;
158  totalread++;
159  }
160  } else {
161  // EOF detected
162  break;
163  }
164  }
165  return totalread;
166 }
