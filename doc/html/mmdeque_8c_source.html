157 MMA_HANDLE* mmdq_create(const char* dequename, ushort item_size, ushort nitems) {
158  MMA_HANDLE* mmahp = NULL;
159  char tagbuff[MAX_DEQUE_NAME_LEN];
160  char* dequefile;
161  size_t len;
162  DQHEADER* dequep = NULL;
163  size_t buffx = 0;
164 
165  memset(tagbuff, 0, sizeof(tagbuff));
166  strncpy(tagbuff, dequename, sizeof(tagbuff)-1);
167  dequefile = mmdq_dequepath(NULL, dequename);
168  len = deque_file_len(item_size, nitems);
169  // printf("Dequefile name = %s\n", dequefile);
170  mmahp = mmapfile_create(tagbuff, dequefile, len, MMA_READ_WRITE, MMF_SHARED, 0664);
171  free(dequefile); // release the dequefile string buffer
172  if (mmahp == NULL) {
173  return NULL; // Error ... let application handle it.
174  }
175  // Now get the data pointer from the MMA_HANDLE. We're going to format
176  // a deque header at that address. We will calculate the buffer offset relative
177  // to the header and set up a deque on the memory mapped region.
178  dequep = (DQHEADER*)mma_data_pointer(mmahp);
179  buffx = buffer_start_offset(dequep);
180 
181  // Now initialize the memory mapped deque.
182  dq_init_memmap(nitems, item_size, buffx, dequep);
183 
184  return mmahp;
185 }

451 void* mmdq_read_packet(MMA_HANDLE* mmdqhp, size_t* packet_lenp) {
452  MMDQ_PACKET_HEADER header;
453  void* packetp;
454  DQSTATS statsbuff;
455  DQSTATS* dqstatsp;
456  int i;
457 
458  dqstatsp = mmdq_stats(mmdqhp, &statsbuff);
459  *packet_lenp = 0;
460  if (dqstatsp->dqitem_size != sizeof(unsigned char)) {
461  APP_ERR(stderr, "For use as packet deque, deque item size must be size of single byte!");
462  }
463 
464  if (pop_packet(mmdqhp, &header, sizeof(MMDQ_PACKET_HEADER))) {
465  DBG_TRACE(stderr, "Error reading packet header from packet deque at %s", mma_get_disk_file_path(mmdqhp));
466  return NULL;
467  }
468  packetp = calloc(header.packet_len, sizeof(void*));
469  if (! sync_check(&header)) {
470  DBG_TRACE(stderr, "mmdq_read_packet detected packet sync issue!");
471  }
472  for (i = 0; i < header.packet_len; i++) {
473  if (pop_packet(mmdqhp, packetp, header.packet_len)) {
474  DBG_TRACE(stderr, "Error reading packet data from packet deque at %s", mma_get_disk_file_path(mmdqhp));
475  free(packetp);
476  return NULL;
477  }
478  }
479  return packetp;
480 }
