51 static void set_dirpath(PATHINFO_STRUCT* pathinfop, PDQHEADER dequep) {
52  char* cp;
53  DQSTATS statsbuff;
54  DQSTATS* dq_statsp;
55 
56  // Allocate a suitably sized buffer for the directory
57  // path part of the pathname. This has been loaded
58  // into the deque.
59  dq_statsp = dq_stats(dequep, &statsbuff);
60  if (dq_statsp->dquse > 0) {
61  pathinfop->dirpath = (char*)calloc(dq_statsp->dquse + 1, sizeof(char));
62  cp = pathinfop->dirpath;
63 
64  // Pop chars from front of queue and copy into the buffer.
65  while (! dq_rtd(dequep, cp)) {
66  cp++;
67  }
68  }

87 static void parse_filename(PATHINFO_STRUCT* pathinfop, char* filename) {
88  char* buff;
89  char* fn;
90  char* ext;
91  int ishiddenfile = 0;
92  size_t fnbufflen;
93  char* fnbuff;
94 
95  if ((filename == NULL) || (strlen(filename) == 0)) {
96  return;
97  }
98  buff = strdup(filename);
99  if (filename[0] == '.') {
100  // Hidden file with leading period
101  ishiddenfile = 1;
102  }
103  fn = strtok(buff, ".");
104  if (fn != NULL) {
105  fnbufflen = strlen(filename);
106  fnbuff = (char*)calloc(fnbufflen, sizeof(char));
107  ext = findext(fnbuff, fn);
108  if (ext != NULL) {
109  pathinfop->extension = strdup(ext);
110  }
111  if (ishiddenfile) {
112  char* fnbasebuff;
113 
114  fnbasebuff = (char*)calloc(strlen(fn) + 2, sizeof(char));
115  strcpy(fnbasebuff, ".");
116  strcat(fnbasebuff, fnbuff);
117  pathinfop->basename = fnbasebuff;
118  } else {
119  pathinfop->basename = strdup(fnbuff);
120  }
121  pathinfop->filename = strdup(filename);
122  }
123  free(buff);
124 }

248 /*
249  * Parse a file path and return a structure containing its
250  * components.
251  *
252  * This function takes a file path (e.g. /var/tmp/myexample.txt)
253  * and produces a structure containing points to the path elements.
254  *
255  * typedef struct _PATHINFO_STRUCT {
256  * char* fullpath;
257  * char* filename;
258  * char* dirpath;
259  * char* extension;
260  * int isdir;
261  * } PATHINFO_STRUCT;
262  *
263  * The PATHINFO_STRUCT structure is allocated from the heap, as are
264  * the strings pointed two by its members. This memory must be
265  * returned to the heap by calling pathinfo_release, passing a pointer
266  * to the PATHINFO_STRUCT returned by pathinfo_parse.
267  */
268 PATHINFO_STRUCT* pathinfo_parse_filepath(const char* pathname) {
269  DQHEADER deque;
270  PATHINFO_STRUCT* pathinfop;
271  char* pathdelim = "/";
272  char* pathbuff;
273  char* tokenp = NULL;
274  char* token_aheadp = NULL;
275  int isdir = 0;
276 
277  // Get a pathinfo structure
278  pathinfop = calloc(1, sizeof(PATHINFO_STRUCT));
279  pathinfop->fullpath = strdup(pathname);
280  // Initialize a plenty big deque to hold path components.
281  dq_init(strlen(pathname) + 1, sizeof(char), &deque);
282 
283  // Break the file path into tokens. Copy to a working buffer
284  // so we don't alter path with calls to strtok
285  pathbuff = strdup(pathname);
286 
287  // Check for leading /
288  if (pathbuff[0] == '/') {
289  push_token(&deque, "/");
290  }
291  isdir = pathinfo_is_dir(pathname);
292  pathinfop->isdir = isdir;
293 
294  // Get first token and look ahead token
295  tokenp = strtok(pathbuff, pathdelim);
296  token_aheadp = strtok(NULL, pathdelim);
297  while (tokenp != NULL) {
298  if (token_aheadp != NULL) {
299  // More tokens in path string
300  push_token(&deque, tokenp);
301  push_token(&deque, "/");
302  tokenp = token_aheadp;
303  token_aheadp = strtok(NULL, pathdelim);
304  } else {
305  // tokenp is the last token in the sequence
306  if (!isdir) {
307  parse_filename(pathinfop, tokenp);
308  } else {
309  push_token(&deque, tokenp);
310  }
311  tokenp = NULL;
312  }
313  }
314  set_dirpath(pathinfop, &deque);
315  free(pathbuff);
316  return pathinfop;
317 }

382 /*
383  * Utility function for appending two elements together to make a complete
384  * path. The returned buffer is allocate from the heap and must be returned
385  * by free.
386  *
387  * The path written will be of the form <path element11>/<path element2>
388  * e.g.
389  *
390  * pelement1 = "/usr/local"
391  * pelement2 = "/etc/foo.conf"
392  *
393  * produces /usr/local/foo.conf
394  *
395  */
396 char* pathinfo_append2path(const char* pelement1, const char* pelement2) {
397  char* pathbuff;
398 
399  pathbuff = (char*)calloc((strlen(pelement1) + strlen(pelement2) + 3), sizeof(char));
400  strcpy(pathbuff, pelement1);
401  if (pathbuff[strlen(pelement1) - 1 ] != '/') {
402  strcat(pathbuff, "/");
403  }
404  strcat(pathbuff, pelement2);
405  return pathbuff;
406 }
