102 int cmdarg_register_option(char* opt_name, char* verbose_name,
103  CA_MODE mode, char* description, char* default_value, char* parent_opt_name) {
104 
105  // 1. Verify this option has not yet been defined.
106  if (find_option(&root, opt_name)) {
107  fprintf(stderr, "Option %s has already been registered\n", opt_name);
108  register_error = 1;
109  return 1;
110  }
111 
112  // 2. If parent_option is null, add to the root option. Otherwise, search for
113  // the named option.
114  if (NULL == parent_opt_name) {
115  add_option(&root, opt_name, verbose_name, mode, description, default_value);
116  } else {
117  CMD_ARG* optionp;
118 
119  optionp = find_option(&root, parent_opt_name);
120  if (NULL == optionp) {
121  fprintf(stderr, "Error registering child option %s -- Parent Option %s has not been registered!\n",
122  opt_name, parent_opt_name);
123  register_error = 1;
124  return register_error;
125  }
126  add_option(optionp, opt_name, verbose_name, mode, description, default_value);
127  }
128  return 0;
129 }

141 int cmdarg_parse(int argc, char* argv[]) {
142  int iarg;
143  int parse_error = 0;
144  char* valp;
145  CMD_ARG* argp;
146  char opt_text[MAX_OPTION_NAME_LEN+1];
147  char* opt_textp;
148 
149  if (parsing_complete) {
150  // We've already parsed these arguments ... no need to do it again
151  // This supports standard definitions of command line args by
152  // ulppk components, like socketserver.c
153  return 0;
154  }
155 
156  if (register_error) {
157  fprintf(stderr, "Command Line Option Parse Error: Option registration error previously detected\n");
158  fprintf(stderr, "Program %s aborting!\n", argv[0]);
159  exit(1);
160  }
161 
162  // Mark all options defined as immediate children of
163  // the root (program name) as enabled. After all,
164  // the program name has clearly been provided.
165  mark_child_options(&root);
166  for (iarg = 1; iarg < argc && !parse_error; iarg++) {
167  opt_textp = clean_option_text(opt_text, argv[iarg]);
168  if (opt_textp == NULL) {
169  // Did not find option when expected ... flag warning and continue loop
170  fprintf(stderr, "Command Line Option Parse Error:"
171  "Found text %s when expecting an option\n", argv[iarg]);
172  continue;
173  }
174  argp = find_option(&root, opt_textp);
175  if (argp == NULL) {
176  fprintf(stderr, "Command Line Option Parse Error: Option %s not registered\n",
177  argv[iarg]);
178  parse_error = 1;
179  break;
180  }
181  // We found a registered option that matches the command line argument
182  switch (argp->mode) {
183  case CA_SWITCH:
184  argp->option_provided = 1;
185  break;
186  case CA_DEFAULT_ARG:
187  case CA_OPTIONAL_ARG:
188  if (iarg+1 >= argc) {
189  fprintf(stderr, "Command Line Option Parse Error:\n"
190  "Command Line Missing Value for Option %s\n", argv[iarg]);
191  parse_error = 1;
192  } else {
193  if (valid_option(argv[iarg+1])) {
194  // Next cmd line argument is an option parse error
195  fprintf(stderr, "Command Line Option Parse Error:"
196  "Option %s requires value before option %s\n",
197  argv[iarg], argv[iarg+1]);
198  parse_error = 1;
199  } else {
200  // Next cmd line argument is option value ... use it
201  argp->option_provided = 1;
202  valp = (char*)calloc(strlen(argv[iarg+1])+1, 1);
203  strcpy(valp, argv[iarg+1]);
204  argp->opt_value = valp;
205  iarg++;
206  }
207  }
208  break;
209  case CA_REQUIRED_ARG:
210  if (iarg+1 >= argc) {
211  fprintf(stderr, "Command Line Option Parse Error:\n"
212  "Command Line Missing Value for Required Option %s\n", argv[iarg]);
213  parse_error = 1;
214  } else {
215  if (valid_option(argv[iarg+1])) {
216  // Next cmd line argument is an option parse error
217  fprintf(stderr, "Command Line Option Parse Error:"
218  "Required Option %s requires value before option %s\n",
219  argv[iarg], argv[iarg+1]);
220  parse_error = 1;
221  } else {
222  // Next cmd line argument is option value ... use it
223  argp->option_provided = 1;
224  valp = (char*)calloc(strlen(argv[iarg+1])+1, 1);
225  strcpy(valp, argv[iarg+1]);
226  argp->opt_value = valp;
227  iarg++;
228  }
229  }
230  break;
231  case CA_ROOT:
232  fprintf(stderr, "Unexpected visitation of root node!\n");
233  break;
234  }
235  // Now visit any child nodes of this node and mark them to indicate
236  // the parent option has been provided. This allows required
237  // options of "submenus" to be enforced ONLY when the parent
238  // option has been provided.
239  mark_child_options(argp);
240  }
241  if (!verify_required_args(&root)) {
242  parse_error = 1;
243  } else {
244  parsing_complete = 1;
245  }
246  return parse_error;
247 }

421 long cmdarg_fetch_long(CMD_ARG* argp, char* opt_name) {
422  long x;
423  CMD_ARG* xargp;
424  char* valp = NULL;
425 
426  if (NULL == argp) {
427  argp = &root;
428  }
429  xargp = find_option(argp, opt_name);
430  if ((xargp->opt_value == NULL) || (strlen(xargp->opt_value) == 0)) {
431  if (xargp->mode == CA_DEFAULT_ARG) {
432  valp = xargp->default_value;
433  }
434  } else {
435  valp = xargp->opt_value;
436  }

438  fprintf(stderr, "ABORTING: NULL or zero length option value when expecting integer\n"
439  "Option is %s (%s)\n",
440  xargp->opt_name, xargp->opt_vname
441  );
442  exit(1);
443  }
444  if (valp != NULL) {
445  errno = 0;
446  x = strtol(valp, NULL, 0);
447  if (errno != 0) {
448  fprintf(stderr, "ABORTING: Unable to parse option value as integer\n"
449  "Option is %s (%s), value is %s errno = %d (%s)\n",
450  xargp->opt_name, xargp->opt_vname, xargp->opt_value, errno,
451  strerror(errno)
452  );
453  exit(1);
454  }
455  }
456  return x;
457 }

582 static int add_option(CMD_ARG* argp, char* opt_name, char* opt_vname, CA_MODE mode,
583  char* description, char* default_value) {
584 
585  CMD_ARG *newargp;
586  PLL_NODE nodep;
587 
588  // If parent is NULL, presume root
589  if (NULL == argp) {
590  argp = &root;
591  }
592 
593  // Construct the new argument node.
594 
595  newargp = (CMD_ARG*)calloc(sizeof(CMD_ARG), 1);
596  newargp->opt_name[0] = opt_name[0];
597  if ((opt_vname != NULL) && (strlen(opt_vname) > 0)) {
598  strncpy(newargp->opt_vname, opt_vname, sizeof(argp->opt_vname) - 1);
599  }
600  if ((description != NULL) && (strlen(description) > 0)) {
601  newargp->description = (char*)calloc(strlen(description) + 1, 1);
602  strcpy(newargp->description, description);
603  }
604  newargp->mode = mode;
605  if ((default_value != NULL) && (strlen(default_value) > 0)) {
606  newargp->default_value = (char*)calloc(strlen(default_value) + 1, 1);
607  strcpy(newargp->default_value, default_value);
608  }
609  newargp->parentp = argp;
610 
611  // Add to the list of parent node options
612 
613  nodep = (PLL_NODE)calloc(sizeof(LL_NODE), 1);
614  nodep->data = newargp;
615  ll_addback(&argp->options, nodep);
616  return 0;
617 }
