183 PLL_NODE ll_getfront(PLL_HEAD header) {
184 
185  PLL_NODE node; /* node retrieved from list */
186 
187  if (header->ll_fptr == NULL ) { /* Then list is empty */
188  node = NULL; /* return with NULL pointer */
189  } else { /* list has at least one item */
190  node = header->ll_fptr; /* get ptr to front object */

196  header->ll_cnt--;

212 PLL_NODE ll_getback(PLL_HEAD header) {
213 
214  PLL_NODE node, pred;
215 
216  if (header->ll_fptr == NULL ) { /* list is empty */
217  node = NULL;
218  } else {
219  node = header->ll_bptr; /* get ptr to node on back of list */
220  if (header->ll_fptr == header->ll_bptr) { /* only one item on list */

291 unsigned char ll_pluck(PLL_HEAD header, PLL_NODE node) {
292  PLL_NODE predecessor; /* predecessor to target node */
293  unsigned char flag;
294 
295  /*
296  * Test for special case 1: node at front of list
297  */
298 
299  if (header->ll_fptr == node) {
300  flag = FALSE; /* clear error flag */
301  ll_getfront(header); /* pull the node off the front of the list */
302  } else {
303  /*
304  * Test for special case 2: node at back of list
305  */
306  if (header->ll_bptr == node) {
307  flag = FALSE; /* clear error flag */
308  ll_getback(header); /* pull the node off the back of the list */
309  } else {
310  /*
311  * Node is either in the middle of the list or not present
312  */
313  predecessor = ll_search(header, (char*) node, ll_match_link);
314 
315  if (predecessor == NULL ) {
316  /*
317  * Node is not on list ... set flag accordingly
318  */
319  flag = TRUE;
320  } else {
321  /*
322  * Node is on list ... set flag and remove it
323  */
324  flag = FALSE; /* clear error flag */
325  ll_plu0(predecessor); /* remove node from list */
326  }
327  }
328  }
329  return (flag);

455 PLL_NODE ll_search(PLL_HEAD header, void* match_ptr,
456  unsigned char (*match_func)(void*, PLL_NODE)) {
457 
458  PLL_NODE cptr; /* pointer to current node */
459  unsigned char match; /* match detected flag */
460 
461  match = FALSE; /* initialize match detected flag */
462  cptr = header->ll_fptr; /* point to object at front of list */
463  while ((cptr != NULL )&& (!match)){ /* while more objects to scan */
464  match = (*match_func)(match_ptr,cptr); /* check for a match */
465  if (!match) { /* match not detected */
466  cptr = cptr->link; /* get next node on list */
467  }
468  }
469  /*
470  * At this point, we are either at the end of the list or we have found
471  * the first node on the list which matches the search criteria.
472  */
473  return (cptr);
474 }
