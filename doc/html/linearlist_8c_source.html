175 void* linlist_add_record(MMFOR_HANDLE* mmforhp, void* userp) {
176  int x;
177  LINEAR_LIST_HEADER* headerp;
178  void* recp;
179 
180  // Lock access
181  linlist_list_lock(mmforhp);
182 
183  // Get the header record
184  headerp = (LINEAR_LIST_HEADER*)mmfor_x2p(mmforhp, 0);
185 
186  if (headerp->nextx > headerp->capacity) {
187  recp = NULL;
188  } else {
189  recp = mmfor_x2p(mmforhp, headerp->nextx);
190  memcpy(recp, userp, headerp->rec_size);
191  headerp->nextx++;
192  }
193  // Unlock access
194  linlist_list_unlock(mmforhp);
195  return recp;
196 }

218 int linlist_delete_recordx(MMFOR_HANDLE* mmforhp, size_t x) {
219  LINEAR_LIST_HEADER* headerp;
220  size_t fence;
221  void* pfence;
222  void* pdelrec;
223  int retval = 0;
224 
225  // Lock access to table
226  linlist_list_lock(mmforhp);
227 
228  headerp = (LINEAR_LIST_HEADER*)mmfor_x2p(mmforhp, 0);
229  fence = headerp->nextx - 2; // -1 to get high index, another -1 to account for header
230  if (fence < 0) {
231  // List is empty. Can't delete
232  ULPPK_LOG(ULPPK_LOG_ERROR, "list is empty ... cannot delete");
233  retval = 1;
234  } else if (x > fence) {
235  // Invalid record index
236  ULPPK_LOG(ULPPK_LOG_ERROR, "Invalid record index: recordx = %d fence = %d",
237  x, fence);
238  retval = 1;
239  } else if (fence == x) {
240  // Deleting record at the fence. Just decrement nextx
241  headerp->nextx -= 1;
242  ULPPK_LOG(ULPPK_LOG_DEBUG, "Deleted fence record [%d]", x);
243  } else {
244  // Copy record at the fence to record index x and dec nextx
245  pdelrec = linlist_x2p(mmforhp, x);
246  pfence = linlist_x2p(mmforhp, fence);
247  memcpy(pdelrec, pfence, headerp->rec_size);
248  ULPPK_LOG(ULPPK_LOG_DEBUG, "Deleted recordx [%d] moved record [%d]",
249  x, fence);
250 
251  headerp->nextx -= 1;
252  }
253  linlist_list_unlock(mmforhp);
254  return retval;
255 }
