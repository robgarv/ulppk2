194  FILE* f;
195  char* rec;

197 
198  if (NULL == trapfile) {

204  trapfile, errno, strerror(errno));
205  return NULL;
206  }
207  rec = fgets(recbuff, sizeof(recbuff), f);
208  while (NULL != rec) {
209  strip_commentary(rec);
210  if (strstr(rec, trap_name) != NULL) {
211  fclose(f);
212  return rec;
213  }
214  rec = fgets(recbuff, sizeof(recbuff), f);
215  }
216  fclose(f);
217  return NULL;
