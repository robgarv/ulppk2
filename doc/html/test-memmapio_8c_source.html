120 static int process_switch_testa() {
121  char fpathbuff[512];
122  int status = 0;
123  int recx = 0;
124  char* strdir = NULL;
125  MMFOR_HANDLE* mmfhp = NULL;
126  TESTA_REC* recp = NULL;

129  status = 1;

131  // Construct deque to hold TA_REC for readback test
132  dq_init(TA_NRECS, sizeof(TESTA_REC), &deque);
133 
134  if (cmdarg_fetch_switch(NULL, "a")) {
135  fprintf(stdout, "TEST-A: File of Records Test\n");
136  strdir = cmdarg_fetch_string(NULL, "d");
137  if (NULL == strdir) {
138  fprintf(stdout, "TEST-A: Target directory not provided in cmd args\n");
139  exit(1);
140  }
141  sprintf(fpathbuff,"%s/TEST-A.DAT",strdir);
142  fprintf(stdout, "Writing file %s\n", fpathbuff);
143  mmfhp = mmfor_create(fpathbuff, MMA_READ_WRITE, MMF_SHARED, 0660,
144  sizeof(TESTA_REC), TA_NRECS);
145  for (recx = 0; recx < TA_NRECS; recx++) {
146  char txtbuff[sizeof(recp->data)];
147 
148  recp = (TESTA_REC*)mmfor_x2p(mmfhp, recx);
149  if (NULL == recp) {
150  fprintf(stdout, "TEST-A Fails! NULL pointer from mmfor_x2p at index %d\n",
151  recx);
152  exit(1);
153  }
154  sprintf(txtbuff, "REC-%d", recx);
155  write_testa_rec(recp, txtbuff, recx);
156  fprintf(stdout, "Data: [%s]\n", recp->data);
157  dq_abd(&deque, recp);
158  }
159  fprintf(stdout, "\nWRITE TEST COMPLETE\n");
160  fprintf(stdout, "Closing test file %s\n", fpathbuff);
161  mmfor_close(mmfhp);
162  fprintf(stdout, "Re-opening test file %s\n", fpathbuff);
163  mmfhp = NULL;
164  mmfhp = mmfor_open(fpathbuff, MMA_READ_WRITE, MMF_SHARED);
165  if (NULL == mmfhp) {
166  fprintf(stdout, "TEST-A: mmfor_open fails!\n");
167  exit(1);
168  }
169  for (recx= 0; recx < TA_NRECS; recx++) {
170  TESTA_REC xrec;
171 
172  dq_rtd(&deque, &xrec);
173  recp = (TESTA_REC*)mmfor_x2p(mmfhp, recx);
174  if (cmp_testa_rec(&xrec, recp)) {
175  fprintf(stdout, "TEST-A: Readback compare fails at record %d\n", recx);
176  exit(1);
177  }
178  fprintf(stdout, "Read Data: [%s]\n", recp->data);
179  }
180  dq_close(&deque);
181  fprintf(stdout, "TEST-A: Completed for %d records\n", TA_NRECS);
182  return status;
183  }
184  return status;
185 }

249 unsigned long parse_index(const char* s) {
250  static char* delim = "|";
251  char* buff;
252  char *pToken;
253  int val = 0;
254  if (strstr(s, delim) == NULL) {
255  val = 0;
256  } else {
257  buff = (char*)calloc(strlen(s) + 1, 1);
258  strcpy(buff, s);
259  pToken = strtok(buff, delim);
260  val = atol(pToken);
261  free(buff);
262  }
263  return val;

352 int main(int argc, char* argv[]) {
353  char* pargv[] = {"a", "b", "c"};
354  static int switches[] = {'h', 'a', 'b', 'c', '\0'};
355  static int (*process_func[])() = {
356  process_switch_help,
357  process_switch_testa,
358  process_switch_testb,
359  process_switch_testc,
360  NULL
361  };
362  int status = 0;
363  int i;
364 
365  flog = stdout;
366  if (argc == 1) {
367  // No arguments supplied ...presume
368  // called from make check we want to execute all tests.
369  register_args(4, pargv);
370  } else {
371  // We have command line arguments
372  register_args(argc, argv);
373  }
374 
375  status = cmdarg_parse(argc, argv);
376  if (status) {
377  cmdarg_show_help(NULL);
378  exit(1);
379  }
380 
381  // Parsed arguments successfully ... process. Handler routines
382  // return > 0 if successful, 0 if they took no action, and < 0
383  // if they encountered an error.
384  for (status = 0, i = 0; (status >= 0 && switches[i] != 0); i++) {
385  status = (*process_func[i])();
386  }
387  return status;
388 }
