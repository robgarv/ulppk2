194 PROC_STATUS* proc_throttled_wait(FCDS* fcdsp, PROC_STATUS* proc_statusp, int options, int wait4pid){
195  pid_t pid;
196  pid_t pid2waitfor = -1;
197  PROC_STATUS* list_proc_statusp;
198  int procstat_allocated = 0;
199 
200  if (NULL == proc_statusp) {
201  proc_statusp = proc_new_proc_status();
202  procstat_allocated = 1;
203  } else {
204  if (wait4pid) {
205  pid2waitfor = proc_statusp->pid;
206  }
207  }
208 
209  pid = waitpid(pid2waitfor, &proc_statusp->status, options);
210  if (pid <= 0) {
211  // Either an error or WNOHANG set and no child terminated
212  // since last call to waitpid
213  proc_statusp->pid = pid;
214  }
215  while (pid > 0) {
216  // A child has terminated. It may be a managed child
217  // or an unmanaged child. Find the corresponding PROC_STATUS
218  // record in the FCDS list, if one exists. If no such status
219  // record is found, then we are dealing with an unmanaged child.
220  list_proc_statusp = search_procs(fcdsp, pid);
221  if (NULL != list_proc_statusp) {
222  // An managed child process terminated ... we have to manage throttle state
223  // Decrement the process count under throttle rules. Then copy necessary
224  // fields to proc_statusp and free list_proc_statusp.
225  fcds_throttle_decrement(fcdsp);
226  proc_statusp->pid = pid;
227  proc_statusp->start_t = list_proc_statusp->start_t;
228  proc_statusp->timeout_sec = list_proc_statusp->timeout_sec;
229  proc_statusp->datap = list_proc_statusp->datap;
230  proc_free_proc_status(list_proc_statusp);
231  pid = 0; // to force loop termination
232  } else {
233  // Non managed child process terminated. If options set WNOHANG, we
234  // can return with a 0 pid. Otherwise, we should wait again. This
235  // one of those cases where use of a goto might actually provide
236  // superior program structure ...
237  if (!(options & WNOHANG)) {
238  pid = waitpid(pid2waitfor, &proc_statusp->status, options);
239  if (-1 == pid) {
240  ULPPK_LOG(ULPPK_LOG_ERROR, "-1 returned from waitpid unexepectedly!");
241  if (procstat_allocated) {
242  free(proc_statusp);
243  }
244  proc_statusp = NULL;
245  }
246  } else {
247  pid = 0;
248  proc_statusp->pid = 0;
249  }
250  }
251  }
252  return proc_statusp;
253 }

298 int proc_make_daemon(char* workingdir, int prohibit_ctty) {
299  pid_t pid;
300  char* wdir;
301  struct stat statbuf;
302  int status;
303  int ok2chdir = 1;
304  int chdir_result = 0;
305 
306  if ((pid = fork()) < 0) {
307  ULPPK_LOG(ULPPK_LOG_ERROR, "Fork error (make daemon)");
308  return -1; // fork error
309  } else if (pid != 0) {
310  exit(0); // parent (1) immediately exits
311  }
312  if (prohibit_ctty) {
313  // Fork again and cause parent (2) to exit
314  if ((pid = fork()) < 0) {
315  ULPPK_LOG(ULPPK_LOG_ERROR, "Fork error (make daemon prohibiting ctty");
316  exit(1); // gotta bail
317  } else if (pid == 0) {
318  // This is parent(2). Exit
319  exit(0);
320  }
321  }
322 
323  // Continue as daemon. If prohibit_ctty is false, then we are a
324  // session leader of a new session.
325  if (NULL == workingdir) {
326  chdir_result = chdir("/");
327  } else {
328  status = stat(workingdir, &statbuf);
329  if (status) {
330  ULPPK_LOG(ULPPK_LOG_ERROR, "Attempting to stat working dir %s", workingdir);
331  ULPPK_LOG(ULPPK_LOG_ERROR, "Forcing working dir to /");
332  chdir_result = chdir("/");
333  ok2chdir = 0;
334  }
335  if (!S_ISDIR(statbuf.st_mode)) {
336  ULPPK_LOG(ULPPK_LOG_ERROR, "Specified working dir is not a directory %s", workingdir);
337  ULPPK_LOG(ULPPK_LOG_ERROR, "Forcing working dir to /");
338  chdir_result = chdir("/");
339  ok2chdir = 0;
340  }
341  if (ok2chdir){
342  chdir_result = chdir(workingdir);
343  }
344  }
345  umask(0);
346  return 0;
347 }

354 static PROC_STATUS* spawn_proc(FCDS* fcdsp, PROC_STATUS* proc_statusp) {
355  pid_t pid;
356 
357  pid = fork();
358  if (pid < 0) {
359  ULPPK_CRASH("Fork error!");
360  } else if (pid == 0) {
361  // This is the child. Return a NULL PROC_STATUS pointer
362  proc_statusp->pid = pid;
363  } else {
364  // This is the parent.
365  proc_statusp->pid = pid;
366  proc_statusp->status = 0;
367  time(&proc_statusp->start_t);
368 
369  // Perform throttle incremenet. This will add
370  // a copy of the PROC_STATUS structure to the
371  // list of managed processes.
372  fcds_throttle_increment(fcdsp, proc_statusp);
373  }
374  return proc_statusp;
375 }
376 static void fcds_throttle_decrement(FCDS* fcdsp) {
377  if (fcdsp->proc_count > 0) {
378  fcdsp->proc_count--;
379  } else {
380  ULPPK_LOG(ULPPK_LOG_DEBUG, "FCDS process count at zero and throttle decrement invoked");
381  }
382  if (fcdsp->proc_count < fcdsp->throttle_exit) {
383  if (fcdsp->throttle_flag) {
384  fcdsp->throttle_flag = 0;
385  ULPPK_LOG(ULPPK_LOG_DEBUG, "Throttle Exit");
386  }
387  }

389 static void fcds_throttle_increment(FCDS* fcdsp, PROC_STATUS* proc_statusp) {
390  if (fcdsp->proc_count < fcdsp->proc_limit) {
391  // Add the proc status by incrementing proc_count and
392  // pushing onto the list
393  fcdsp->proc_count++;
394  dq_abd(&fcdsp->procs, proc_statusp);
395  proc_dump_proc_status(NULL, proc_statusp);
396  }
397 
398  if (fcdsp->proc_count >= fcdsp->proc_limit) {
399  if (!fcdsp->throttle_flag) {
400  ULPPK_LOG(ULPPK_LOG_WARN, "Throttle Entry at proc_count %d", fcdsp->proc_count);
401  fcdsp->throttle_flag = 1;
402  }
403  }

405 static PROC_STATUS* search_procs(FCDS* fcdsp, pid_t pid) {
406  DQSTATS dqstats;
407  DQSTATS* dqstatsp;
408  PROC_STATUS list_proc_status;
409  PROC_STATUS* outproc_statusp = NULL;
410  DQHEADER* proc_listp;
411  int i;
412  int save_key;
413 
414  proc_listp = &fcdsp->procs;
415  dqstatsp = dq_stats(proc_listp, &dqstats);
416  if (dqstatsp->dquse != fcdsp->proc_count) {
417  ULPPK_LOG(ULPPK_LOG_WARN, "FCDS proc list size != FCDS proc count: VERY UNEXPECTED");
418  }
419  for (i = 0; i < dqstatsp->dquse; i++) {
420  memset(&list_proc_status, 0, sizeof(PROC_STATUS));
421  if (!dq_rtd(proc_listp, &list_proc_status)) {
422  // Successful pop from top of deque
423  proc_dump_proc_status(NULL, &list_proc_status);
424  if (list_proc_status.pid == pid) {
425  // Found a matching record. We will copy its contents
426  // to a new structure and let it fall of the deque.
427  outproc_statusp = proc_new_proc_status();
428  // Now we copy everything from list_proc_status to outproc_statusp EXCEPT
429  // for the key field. outproc_statusp->key must be PROC_STATUS_GUARD1.
430  save_key = outproc_statusp->key;
431  memcpy(outproc_statusp, &list_proc_status, sizeof(PROC_STATUS));
432  outproc_statusp->key = save_key;
433  return outproc_statusp;
434  } else {
435  // Not our target process .. push back onto the bottom of the deque
436  dq_abd(proc_listp, &list_proc_status);
437  }
438  } else {
439  // this should never happen. If it does, log it and break out of here.
440  ULPPK_LOG(ULPPK_LOG_ERROR, "FCDS list empty when it should have items ... error");
441  break;
442  }
443  }
444  return NULL;
445 }
