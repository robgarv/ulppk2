190 int dq_atd(PDQHEADER deque,void* item) {
191 
192 int flag; // error flag ... TRUE implies error
193 void* lpslot; // ptr to 1st byte of destination slot
194 
195 /* BEGIN */
196 
197 if (!deque->dq_open) { // deque not open
198  return TRUE; // indicate error
199 }
200 if (deque->dqslots > deque->dquse) { // have room in deque
201  if (deque->dquse != 0) { // deque not empty
202  deque->dqtop = (deque->dqtop+1) % deque->dqslots; // adjust top ptr index
203  }
204  deque->dquse++; // increment use count
205  /*
206  * Copy item into slot at slot index deque->dqtop.
207  */
208  lpslot = map_slot(deque,deque->dqtop); // compute ptr to slot
209  memcpy(lpslot,item,deque->dqitem_size); // copy item to deque
210  flag = FALSE; // indicate no error
211 } else { // deque is full
212  flag = TRUE;
213 }
214 
215 return (flag);
216 
217 /* END */
218 
219 }

270 int dq_rtd(PDQHEADER deque,void* item) {
271 
272 int flag; // error flag ... TRUE implies error
273 void* lpslot; // ptr to 1st byte of destination slot
274 int use;
275 
276 /* BEGIN */
277 
278 if (!deque->dq_open) { // deque not open
279  return TRUE; // indicate error
280 }
281 if (deque->dquse > 0) { // deque not empty
282  lpslot = map_slot(deque,deque->dqtop); // compute ptr to slot
283  memcpy(item,lpslot,deque->dqitem_size); // copy item to target area
284  flag = FALSE; // indicate no error
285  /*
286  * Adjust bottom ptr index
287  */
288  if (deque->dquse != 1) {
289  deque->dqtop = (deque->dqtop + deque->dqslots - 1) % deque->dqslots;
290  }
291  flag = FALSE;
292  use = deque->dquse;
293  deque->dquse = use - 1; // decrement slots in use count
294 } else { // attempt to remove from empty deque
295  flag = TRUE;
296 }
297 
298 return (flag);
299 
300 /* END */
301 
302 }
