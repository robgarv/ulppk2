317 int ssrvr_start(SSRVR_HANDLE* pHandle, void* datap, int argc, char* argv[]) {
318  int rstatus = 0;
319  struct sockaddr_in clientaddr;
320  int connfd = 0;
321  int retstat = 0;
322  pid_t childpid;
323 
324  if (pHandle == NULL) {
325  fprintf(stderr, "NULL server handle not allowed! Aborting!\n");
326  return 1;
327  }
328  ssrvrp = pHandle; // set the server handle pointer
329  data_ptr = datap; // set the data pointer
330 
331 
332  // Init command line arg processing and call the
333  // Command Line Personality Function
334  ssrvr_cmdarg_init(argc, argv);
335  if (NULL != ssrvrp->clpf) {
336  rstatus = (*ssrvrp->clpf)(argc, argv, data_ptr);
337  if (rstatus) {
338  fprintf(stderr, "Command Line Personality Function returns %d ... aborting\n",
339  rstatus);
340  return 1;
341  }
342  }
343  // Perform parsing of the command line arguments
344  // This will obtain the values for -m, -p, -b options
345  // (-h, -m, -p and -b options)
346  if (parse_clargs(argc, argv)) {
347  fprintf(stderr, "Error parsing options ... aborting!\n");
348  cmdarg_show_help(NULL);
349  return 1;
350  }
351 
352  if (NULL != ssrvrp->ipf) {
353  rstatus = (*ssrvrp->ipf)(data_ptr);
354  if (rstatus) {
355  fprintf(stderr, "Initialization Personality Function returns %d ... aborting\n",
356  rstatus);
357  return 1;
358  }
359  }
360 
361  // Now set up the listen socket
362  listenfd = setup_listen_socket();
363 
364  // Set child signal handler
365  if (set_signal(SIGCHLD, sig_child_handler) == SIG_ERR) {
366  fprintf(stderr, "Error registering SIGCHLD handler ... Aborting\n");
367  return 1;
368  }
369 
370  // Set termination signal handler
371  if (set_signal(SIGTERM, sig_term_handler) == SIG_ERR) {
372  fprintf(stderr, "Error register SIGTERM handler ... Aborting \n");
373  return 1;
374  }
375 
376  // Accept connections
377 
378  for (loop_abort = 0; !loop_abort; ) {
379  socklen_t clasize;
380 
381  clasize = (socklen_t)sizeof(clientaddr);
382  if (!listenfd_forced_closed) {
383  connfd = accept(listenfd, (struct sockaddr*)&clientaddr, &clasize);
384  } else {
385  break; // break out of the loop
386  }
387  if (connfd < 0) {
388  if (EINTR == errno) {
389  continue;
390  } else if (!listenfd_forced_closed) {
391  fprintf(stderr, "accept error! errno = %d | %s\n", errno, strerror(errno));
392  retstat = errno;
393  break;
394  }
395  }
396 
397  if (listenfd_forced_closed) {
398  break; // break out of the for loop
399  }
400 
401  if (open_conns >= max_open_conns) {
402  // "Throttle" by closing the socket connection.
403  // We'll hang another accept ... but no connection will
404  // be processed until a child process terminates.
405  close(connfd);
406  continue;
407  }
408  open_conns++; // increment counter of open connections
409 
410  if ((childpid = fork()) == 0) {
411  int childstatus;
412  // This is the child
413  close(listenfd);
414 
415  // Invoke the main personality function if it is defined
416  if (NULL != ssrvrp->mpf) {
417  childstatus = (*ssrvrp->mpf)(connfd, data_ptr);
418 
419  } else {
420  childstatus = 0;
421  }
422  // Invoke the child termination personality function
423  if (NULL != ssrvrp->ctpf) {
424  (*ssrvrp->ctpf)(connfd, data_ptr);
425  }
426  exit(childstatus);
427  } else {
428  // This is the parent. Close connfd to decrement reference count
429  close(connfd);
430  }
431  }
432  if (!listenfd_forced_closed) {
433  close(listenfd);
434  }
435  if (NULL != ssrvrp->ptpf) {
436  retstat =(*ssrvrp->ptpf)(data_ptr);
437  }
438  return retstat;
439 
440 }
