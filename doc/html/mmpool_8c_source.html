262 int mmpool_putbuff(BPOOL_HANDLE* bphp, BPCF_BUFFER_REF* buffp) {
263  int error = 0;

265  /*
266  * Search the out of pool deque for the item.
267  */

269  // Found a match ... do the deallocation.

329 unsigned short mmpool_zap_pool(BPOOL_HANDLE* bphp) {
330  BPOOL_INDEX bpx;
331  BPCF_BUFFER_REF* bufrefp;
332  void* vp;
333  size_t user_data_size;
334  int return_count = 0;
335 
336  while (!dq_rtd(&bphp->bpmf_recp->dq_outpool, &bpx)) {
337  bufrefp = mmpool_buffx2refp(bphp, bpx); // get buffer reference
338  vp = mmpool_buffer_data(bufrefp); // get pointer to user data
339  user_data_size = bphp->bpmf_recp->stats.max_data_size; // get the max user data size
340  memset(vp, 0, user_data_size); // zap the user data
341  dq_abd(&bphp->bpmf_recp->dq_inpool, &bpx); // put it back into the pool
342  bphp->bpmf_recp->stats.remaining++; // bump current buffer count
343  return_count++;
344  }
345  return return_count;
346 }

425 static MMA_HANDLE* format_bpmf(
426  MMA_HANDLE* mmahp, // memory mapped atom handle
427  char name[MMPOOL_MAX_POOL_NAME], // Symbolic name of the buffer pool
428  unsigned short bp_id, // Buffer pool ID
429  size_t max_data_size, // max number of user bytes to be written to these buffers
430  unsigned short rqst_capacity, // min number of buffers to allocate
431  unsigned short alloc_capacity // actual number of buffers allocated to the pool
432 ) {
433  void* p0;
434  BPMF_STATS* statsp;
435  BPMF_REC* bpmf_recp;
436  DQHEADER* dqinp;
437  DQHEADER* dqoutp;
438  void* b1p;
439  void* b2p;
440 
441  p0 = (void*)mma_data_pointer(mmahp); // get ptr to first byte of data
442  statsp = (BPMF_STATS*)p0; // get ptr to stats
443  bpmf_recp = (BPMF_REC*)p0; // and ptr to BPMF_REC
444 
445  // Calculate pointers to in pool and out of pool deques
446 
447  dqinp = &bpmf_recp->dq_inpool;
448  dqoutp = &bpmf_recp->dq_outpool;
449 
450  // Calculate pointers into memory mapped area of buffers
451  // for the two deques. "In pool" buffer area comes first ...
452  b1p = ((void*)dqoutp) + sizeof(DQHEADER);
453  b2p = b1p + alloc_capacity * sizeof(BPOOL_INDEX);
454 
455  // Calculate the offsets relative to p0 of these two deque slot buffers
456  bpmf_recp->indqbuffx = (size_t)(b1p - p0);
457  bpmf_recp->outdqbuffx = (size_t)(b2p - p0);
458 
459  // Format the stats record
460  strncpy(statsp->name, name, MMPOOL_MAX_POOL_NAME);
461  statsp->bp_id = bp_id;
462  statsp->rqst_capacity = rqst_capacity;
463  statsp->capacity = alloc_capacity;
464  statsp->max_data_size = max_data_size;
465  statsp->remaining = statsp->capacity;
466 
467  // Initialize the deques.
468  dq_init_memmap(statsp->capacity, sizeof(BPOOL_INDEX), bpmf_recp->indqbuffx, dqinp);
469  dq_init_memmap(statsp->capacity, sizeof(BPOOL_INDEX), bpmf_recp->outdqbuffx, dqoutp);
470 
471  return mmahp;
472 }

475 static MMA_HANDLE* new_bpmf(
476  char name[MMPOOL_MAX_POOL_NAME], // Symbolic name of the buffer pool
477  unsigned short bp_id, // Buffer pool ID
478  size_t max_data_size, // max number of user bytes to be written to these buffers
479  unsigned short rqst_capacity // min number of buffers to allocate
480 ) {
481  static char fpath[1024];
482  size_t bpmf_size = 0;
483  long pagesize;
484  long pages;
485  long remainder;
486  long slop;
487  long itemslop;
488  unsigned short alloc_capacity;
489  MMA_HANDLE* mmahp = NULL;
490  char* bpmf_file_name;
491 
492  pagesize = sysconf(_SC_PAGESIZE);
493 
494  // Calculate size of the Buffer Pool Management File for this pool. This
495  // will be size of the BPMF_REC control structure and the size of two deque
496  // elements arrays.
497  bpmf_size = sizeof(BPMF_REC) + (2 * sizeof(BPOOL_INDEX) * rqst_capacity);
498 
499  // Calculate number of pages and amount left over
500  pages = bpmf_size / pagesize;
501  remainder = bpmf_size % pagesize;
502  slop = pagesize - remainder;
503 
504 #ifndef _MMPOOL_USE_SLOP
505  alloc_capacity = rqst_capacity;
506 #else
507  if (remainder > 0) {
508  pages ++;
509 
510  // Allocate the slop to the two deques
511  itemslop = slop / (2 * sizeof(BPOOL_INDEX));
512  alloc_capacity = rqst_capacity + itemslop;
513  bpmf_size = sizeof(BPMF_REC) + (2 * sizeof(BPOOL_INDEX) * alloc_capacity);
514  }
515 #endif
516 
517  bpmf_file_name = mmpool_bpmf_filename(name);
518  sprintf(fpath, "%s/%s", variables.data_dir, bpmf_file_name);
519  mmahp = mmapfile_create(name, fpath, bpmf_size, MMA_READ_WRITE, MMF_SHARED,
520  0660);
521  if (mmahp != NULL) {
522  // Successful memory mapped file setup. Format the buffer pool
523  // management file.
524  format_bpmf(mmahp, name, bp_id, max_data_size, rqst_capacity, alloc_capacity);
525 
526  }
527  return mmahp;
528 }

548 /*
549  * Return offset to the first byte of user data in a BPCF_BUFFER.
550  * This is independent upon the particular buffer and is a function
551  * of the size of the header.
552  * The result is always aligned to a 4 byte boundary.

554 static off_t bpcf_data_offset() {
555  size_t header_size;
556  off_t offset;
557 
558  header_size = sizeof(BPCF_BUFFER_REF);
559  if (header_size % 4) {
560  int rem;
561 
562  rem = sizeof(BPCF_BUFFER_REF) % 4;
563  header_size += rem;
564  }
565  offset = (off_t)header_size;
566  return offset;
567 }

569 static int allocate_buffers(MMA_HANDLE* bpmfhp, MMFOR_HANDLE* bpcfhp) {
570  size_t ibuff;
571  void* vbuffp;
572  void* vbuffp0;
573  BPMF_STATS* bpmf_statsp;
574  BPMF_REC* bpmf_recp;
575  BPCF_BUFFER_REF buffref;
576  size_t buff_size;
577 
578  bpmf_recp = (BPMF_REC*)mma_data_pointer(bpmfhp);
579  bpmf_statsp = &bpmf_recp->stats;
580  buff_size = bpmf_statsp->max_data_size + bpcf_data_offset();
581  buffref.bp_id = bpmf_statsp->bp_id;
582  buffref.buff_len = buff_size;
583  buffref.sync_word[0] = 0xA4;
584  buffref.sync_word[1] = 0xA4;
585  vbuffp0 = (void*)mmfor_x2p(bpcfhp, 0);
586  vbuffp = vbuffp0;
587  for (ibuff = 0; ibuff < bpmf_statsp->capacity; ibuff++) {
588  buffref.bpindex = ibuff; // set this buffer's index
589  memset(vbuffp, '*' , buff_size);
590  memcpy(vbuffp, &buffref, sizeof(buffref)); // copy our local structure to mapped memory
591  dq_abd(&bpmf_recp->dq_inpool, &ibuff); // add buffer index to in pool deque
592  vbuffp += buff_size;
593  }
594  return 0;
595 }

627 // Search for the specified buffer pool index. If found, pop it off the deque
628 // and return non-zero. If not, return zero.
629 static int search_deque(DQHEADER* dequep, BPOOL_INDEX bpx) {
630  int nitems;
631  int i;
632  int x;
633 
634  nitems = dequep->dquse;
635  for (i = 0; i < nitems; i++) {
636  dq_rtd(dequep, &x); // pop from top
637  if (x == bpx) {
638  return 1; // that's our match ... return TRUE
639  }
640  dq_abd(dequep, &x); // not a match .. push onto bottom
641  }
642  return 0; // No match found ... return FALSE
643 }
