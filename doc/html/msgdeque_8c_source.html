78 /*
79  * Make a dequename from a message cell name. Dequenames will
80  * be of the form prefix-<msg cell name> ... so dequefiles will
81  * be of the form prefix-<msg cellname>.dq
82  *
83  * If buff is NULL, memory is allocated from the heap. In this
84  * case the calling app must free the memory. If the
85  * buffer is too small, ULPPK_CRASH is called to shutdown the
86  * calling application and log this error.
87  */
88 static char* make_name(char* buff, size_t buffsize, const char* name) {
89  int dqnamelen;
90  static char* prefix = "msgdeque-";
91 
92  dqnamelen = strlen(prefix) + strlen(name) + 1;
93  if ((buff != NULL) && (buffsize < dqnamelen)) {
94  // Caller must supply a sufficiently larger buffer
95  ULPPK_CRASH("Buffer too small: Have %d need %d", buffsize, dqnamelen);
96  } else {
97  buff = (char*)calloc(dqnamelen, sizeof(char));
98  }
99  strcpy(buff, prefix);
100  strcat(buff, name);
101  return buff;
102 }

214 static unsigned char* encode_uint(unsigned char buff[], unsigned int datalen) {
215  buff[0] = (datalen & 0x000000FF);
216  buff[1] = (datalen & 0x0000FF00) >> 8;
217  buff[2] = (datalen & 0x00FF0000) >> 16;
218  buff[3] = (datalen & 0xFF000000) >> 24;
219  return buff;
220 }

222 static unsigned int decode_uint(unsigned char* buff) {
223  unsigned int x;
224  unsigned int reg;
225 
226  x = 0;
227  reg = buff[3];
228  reg = reg << 24;
229  x |= reg;
230  reg = buff[2];
231  reg = reg << 16;
232  x |= reg;
233  reg = buff[1];
234  reg = reg << 8;
235  x |= reg;
236  reg = buff[0];
237  x |= reg;
238  return x;
239 }

251  MMA_HANDLE* deque;
252  int retval = 0;
253 
254  /*
255  * This is an atomistic insertion into the deque ...
256  * mmdq_abd locks/unlocks the deque header, so
257  * no further protection is required. Add to the
258  * deque an signal the deque reader
259  */
260  deque = ((MSGDEQUE*)msgcellp->datap)->deque;
261  retval = mmdq_abd(deque, sendp);
262  if (0 == retval) {
263  msgcell_send(msgcellp);
264  }
265  return retval;

277 void* msgdeque_rec(MSGCELL* msgcellp) {
278  int retval = 0;
279  void* rec = NULL;
280  MMA_HANDLE* deque;
281  MMA_HANDLE* lock;
282  DQSTATS dqstats;
283  DQSTATS* dqstatsp = NULL;
284  size_t recordlen = 0;
285 
286  deque = ((MSGDEQUE*)msgcellp->datap)->deque;
287  lock = ((MSGDEQUE*)msgcellp->datap)->lock;
288 
289  while ((NULL == rec ) && (0 == retval)) {
290  // Lock access to the message deque and get the
291  // current dequeue stats
292 
293  mma_lock_atom_write(lock);
294  dqstatsp = mmdq_stats(deque, &dqstats);
295  if (dqstatsp->dquse != 0) {
296  // We have data
297  recordlen = (size_t)dqstatsp->dqitem_size;
298  rec = (void*)calloc(1, recordlen);
299  if (mmdq_rtd(deque, rec) != 0) {
300  // This should never happen. But we can take some action to
301  // possibly recover from the error.
302  free(rec);
303  rec = NULL;
304  ULPPK_LOG(ULPPK_LOG_ERROR, "Error retrieving from message deque: %s", msgcellp->name);
305  }
306  }
307  // Unlock access to the message deque
308  mma_unlock_atom(lock);
309 
310  // If we still don't have a record
311  if (NULL == rec) {
312  // Wait for a signal on the message cell.
313  retval = msgcell_rec(msgcellp);
314  if (retval) {
315  ULPPK_LOG(ULPPK_LOG_ERROR, "Error receiving on message cell [%d / %s",
316  msgcellp->errcode, strerror(msgcellp->errcode));
317  }
318  }
319  }
320  return rec;
321 }

335 int msgdeque_send_byte_stream(MSGCELL* msgcellp, void* pdata, size_t datalen) {
336  unsigned char lenbuff[4];
337  unsigned char* pbyte;
338  MMA_HANDLE* lock;
339  MMA_HANDLE* deque;
340  int i;
341  int retval = 0;
342 
343  pbyte = (void*)lenbuff;
344  encode_uint(lenbuff, datalen);
345  lock = ((MSGDEQUE*)msgcellp->datap)->lock;
346  deque = ((MSGDEQUE*)msgcellp->datap)->deque;
347  mma_lock_atom_write(lock);
348  // Push the 4 bytes of encoded size onto the deque
349  for (i = 0; ((i < sizeof(lenbuff)) && (retval ==0)); i++, pbyte++) {
350  retval = mmdq_abd(deque, pbyte);
351  }
352  // Now push the data
353  if (0 == retval) {
354  pbyte = (unsigned char*)pdata;
355  for (i = 0; ((i < datalen) && (retval ==0)); i++, pbyte++) {
356  retval = mmdq_abd(deque, pbyte);
357  }
358  }
359  mma_unlock_atom(lock);
360 
361  // Send to the message cell (post to semaphore)
362  msgcell_send(msgcellp);
363  return retval;
364 }

379 void* msgdeque_rec_byte_stream(MSGCELL* msgcellp, size_t* bytes_received) {
380  MMA_HANDLE* deque;
381  MMA_HANDLE* lock;
382  DQSTATS dqstats;
383  DQSTATS* dqstatsp = NULL;
384  unsigned char* databuffp = NULL;
385  unsigned char* pbyte = NULL;
386  int i;
387  int retval = 0;
388  int datalen = 0;
389  char sizebuff[4];
390 
391  deque = ((MSGDEQUE*)msgcellp->datap)->deque;
392  lock = ((MSGDEQUE*)msgcellp->datap)->lock;
393 
394  while ((NULL == databuffp) && (0 == retval)) {
395  // lock access to the message deque
396  mma_lock_atom_write(lock);
397 
398  // Get current deque stats
399  dqstatsp = mmdq_stats(deque, &dqstats);
400  if (dqstatsp->dquse >= 4) {
401  // we got some data!
402  // Pop the four bytes of encoded size of the deque
403  for (i = 0; ((i < sizeof(sizebuff)) && (0 == retval)); i++) {
404  retval = mmdq_rtd(deque, &sizebuff[i]);
405  }
406  // Check for error condition
407  if (retval) {
408  ULPPK_LOG(ULPPK_LOG_ERROR, "Error popping encoded size of byte stream deque %s", msgcellp->name);
409  } else {
410  datalen = decode_uint(sizebuff);
411  if (datalen > 0) {
412  databuffp = (unsigned char*)calloc(datalen, sizeof(unsigned char));
413  for (i = 0, pbyte = databuffp; ((i < datalen) && (0 == retval)); i++, pbyte++ ) {
414  retval = mmdq_rtd(deque, pbyte);
415  }
416  if (retval) {
417  free(databuffp);
418  databuffp = NULL;
419  ULPPK_LOG(ULPPK_LOG_ERROR, "Error popping encoded size of byte stream deque %s", msgcellp->name);
420  }
421  }
422  }
423  }
424  mma_unlock_atom(lock);
425 
426  // If after all that, we have no data ... wait
427  // for some data!
428  if (NULL == databuffp) {
429  retval = msgcell_rec(msgcellp);
430  if (retval) {
431  ULPPK_LOG(ULPPK_LOG_ERROR, "Error receiving on message cell [%d / %s",
432  msgcellp->errcode, strerror(msgcellp->errcode));
433  }
434  }
435  }
436  return databuffp;
437 }
