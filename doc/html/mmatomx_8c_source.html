152 
153  if (cmdarg_fetch_switch(NULL, "d")) {

161  if (NULL == mmahp) {
162  char txtbuff[1024];

164  mma_strerror(txtbuff, sizeof(txtbuff)));
165  app_error(errbuff);

167  // Print some statistics
