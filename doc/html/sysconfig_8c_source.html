116 static char* find_inifile(const char* appname, const char* etcdir) {
117  char* fpath;
118  char* fname;
119  char* searchdirs[] = {
120  ".", // current working directory
121  ".", // this slot will be overwritten with SYSCONFIG_ETC
122  "/usr/local/etc",
123  "/etc",
124  NULL,
125  NULL
126  };
127  char* dirp;
128  struct stat statbuff;
129  int isfile;
130  int ix;
131 
132  if ((appname != NULL) && (strlen(appname) > 0)) {
133  // Read environment variable SYSCONFIG_ETC and add to the list of directories
134  // to search (at the top).
135  searchdirs[1] = (char*)etcdir; // write etcdir to search list index 1
136  fname = inifilename(appname);
137  dirp = searchdirs[0];
138  ix = 0;
139  while (dirp != NULL) {
140  fpath = pathinfo_append2path(dirp, fname);
141  isfile = pathinfo_is_file(fpath);
142  if (isfile) {
143  return fpath;
144  }
145  free(fpath);
146  ix++;
147  dirp = searchdirs[ix];
148  }
149  free(fname);
150  }
151 
152  // If that didn't work, try SYSCONFIG_INI_FILE
153  fname = appenv_read_env_var(SYSCONFIG_INI_FILE_NAME);
154  if (NULL == fname) {
155  syslog(LOG_ERR | LOG_LOCAL0, "Environment variable %s not established ... fatal error", SYSCONFIG_INI_FILE_NAME);
156  kill(getpid(), SIGQUIT);
157  }
158  dirp = searchdirs[0];
159  ix = 0;
160  while (dirp != NULL) {
161  fpath = pathinfo_append2path(dirp, fname);
162  isfile = pathinfo_is_file(fpath);
163  if (isfile) {
164  return fpath;
165  }
166  free(fpath);
167  ix++;
168  dirp = searchdirs[ix];
169  }
170  // No ini file found ... return NULL
171  return NULL;
172 }

236  // Logging not previously set up by a call to sysconfig_set_logging
237  // so setup logging with defaults.
238  //
239  // Open the system log. Note that we cannot use ULPPK_LOG or ULPPK_CRASH
240  // until the .INI file has been successfully parsed. Nor is it
241  // safe to log to stderr (we may not have a console at this point!)
242  openlog(appname, LOG_PID, LOG_LOCAL0);
243  }

247  }

257  kill(getpid(), SIGQUIT);

260  if (NULL == inifile_name) {

262  kill(getpid(), SIGQUIT);
263  }
264

266  inifilepath = find_inifile(appname, etcdir_path);
267  if (inifilepath == NULL) {
268  char* label;
269 
270  if ((appname == NULL) || (strlen(appname) ==0)) {
271  syslog(LOG_ERR, "Unable to access inifile %s", SYSCONFIG_INI_FILE_NAME);
272  } else {
273  syslog(LOG_ERR, "Unable to access inifile %s.ini or %s", appname, SYSCONFIG_INI_FILE_NAME);
274  }
275  kill(getpid(), SIGQUIT);
276  }
277  fp = fopen(inifilepath, "r");
278  if (NULL == fp) {
279  syslog(LOG_ERR, "Cannot open inifile %s", inifilepath);
280  kill(getpid(), SIGQUIT);
281  }

287  }
288

290  if (!logging_set) {
291  // Close syslog and call sysconfig_set_logging
292 
293  }

297  }

300  // (.ini settings have precedence over environment settings)

302  // directory.

305  if (NULL == varp) {

308  }

312  varp = if_get_element(inifilep, "environment", "log_dir");
313  if (NULL == varp) {

316  }
317  } else {
318  appenv_set_env_var(SYSCONFIG_LOG_DIR, varp->element.szvalue);
319  }
320 
321  varp = if_get_element(inifilep, "environment", "core_dir");
322  if (NULL == varp) {
323  if (appenv_read_env_var(SYSCONFIG_CORE_DIR) == NULL) {
324  ULPPK_LOG(ULPPK_LOG_WARN, "Inifile: %s Unable to access environment section, core_dir element", inifilepath);
325  }
326  } else {
327  appenv_set_env_var(SYSCONFIG_CORE_DIR, varp->element.szvalue);
328  }
329 
330  varp = if_get_element(inifilep, "environment", "mmdq_dir");
331  if (NULL == varp) {
332  if (appenv_read_env_var(MMDQ_DIR_PATH) == NULL) {

342  }
343  } else {
344  appenv_set_env_var(MMPOOL_ENV_DATA_DIR, varp->element.szvalue);

349  varp = if_get_element(inifilep, "environment", "ld_library_path");
350  if (NULL != varp) {
351  char* env_varp;
352

355  env_varp = getenv("LD_LIBRARY_PATH");
356  if (NULL == env_varp) {

358  varp->element.szvalue);
359  } else {
360  ULPPK_LOG(ULPPK_LOG_INFO, "LD_LIBRARY_PATH override: %s",
361  varp->element.szvalue);
362  }
363  } else {
364  char* env_varp;
365 
366  env_varp = getenv("LD_LIBRARY_PATH");
367  if (NULL == env_varp) {
368  ULPPK_LOG(ULPPK_LOG_INFO, "No LD_LIBRARY_PATH override. LD_LIBRARY_PATH not set");
369  } else {
370  ULPPK_LOG(ULPPK_LOG_INFO, "Using inherited LD_LIBRARY_PATH: %s", env_varp);
371  }
372  }
373  if (inifilepath != NULL) {
374  free(inifilepath);
375  }
376  return inifilep;

443 int sysconfig_set_user(char* user) {
444  struct passwd* passwdp;
445  struct group* group_p;
446  char* group;
447  uid_t uid;
448  gid_t gid;
449  int retval = 0;
450 
451  // Get the passwd structure for user
452 
453  errno = 0;
454  passwdp = getpwnam(user);
455  if (NULL == passwdp) {
456  ULPPK_LOG(ULPPK_LOG_ERROR,
457  "Unable to access user %s in /etc/passwd! Is %s user defined?", user, user);
458  return 1;
459  }
460  // Got the passwd struct for given user. Get the user ID and group ID
461  uid = passwdp->pw_uid;
462  gid = passwdp->pw_gid;
463  group_p = getgrgid(gid);
464  if (NULL == group_p) {
465  ULPPK_LOG(ULPPK_LOG_ERROR,
466  "Unable to access group ID %d in /etc/group! Is group %gid defined?", gid, gid);
467  return 1;
468  }
469  group = group_p->gr_name;
470  ULPPK_LOG(ULPPK_LOG_DEBUG, "User %s uid [%d] Group %s gid [%d]", user, group, uid, gid);
471 
472  // Now do a setgid and setuid. The order of these operations is important.
473  // If we change user ID first, we won't have permission to change group ID ...
474  if (setgid(gid)) {
475  ULPPK_LOG(ULPPK_LOG_ERROR,
476  "Unable to set group ID to $group/%d: errno [%d]/[%s]", group, gid,
477  errno, strerror(errno));
478  return 1;
479  }
480  ULPPK_LOG(ULPPK_LOG_DEBUG, "Changed group ID to %s/[%d]", group, gid);
481  if (setuid(uid)) {
482  ULPPK_LOG(ULPPK_LOG_ERROR,
483  "Unable to set user ID to %s/%d: errno [%d]/[%s]", user, uid,
484  errno, strerror(errno));
485  return 1;
486  }
487  ULPPK_LOG(ULPPK_LOG_DEBUG, "Changed uid to %s/[%d]", user, uid);
488  return 0;
489 }
