54 /*
55  * Logging destination. A bit map of log destinations
56  * Bit 0: (val 1) => standard error
57  * Bit 1: (val 2) => syslog
58  * Bit 0+1: (val 3) => standard error and syslog

68 static void get_log_vars( )
69 {
70  INIFILE_NODE* inifilep;
71  INIFILE_NODE* varp;
72 
73  log_level = ULPPK_LOG_DEBUG;
74  ulppk_corepath = "/var/crash";

76  inifilep = if_get_root();

78  return;
79  }

87

89  if (varp == NULL) {
90  // Can't call ULPPK_LOG from here ... recurses without limit

92  }

96 /*
97  * Maps a ULPPK log level to a syslog priority code
98  */
99 static int syslog_priority(ULPPK_LOG_LEVEL level) {
100  static int pri[] = {
101  LOG_DEBUG,
102  LOG_INFO,
103  LOG_WARNING,
104  LOG_ERR,
105  LOG_CRIT,
106  LOG_CRIT
107  };
108  if ((level > ULPPK_LOG_CRASH_ERROR) || (level < ULPPK_LOG_DEBUG)) {
109  level = ULPPK_LOG_ERROR;
110  }
111  return pri[level];
112 }

114 static void route_log(ULPPK_LOG_LEVEL level, char* text) {
115  if (log_dest & ULPPK_LOGDEST_SYSLOG) {
116  int syslog_pri;
117 
118  syslog_pri= syslog_priority(level);
119  syslog(syslog_pri, "%s", text);
120  }
121  if (log_dest & ULPPK_LOGDEST_CONSOLE) {
122  fprintf(stderr, "%s\n", text);
123  }
124 }
